## require 直接引入文件

    
    
    -- aa.lua文件
    function showName( v )
        -- body
        print ("哦哈呦"..v)
    end
    
    
    
    -- a.lua 文件内引入aa.lua文件
    require "aa"
    showName("n加奈")
