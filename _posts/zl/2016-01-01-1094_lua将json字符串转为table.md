### json字符串转为table

    
    
    local cjson = require "cjson"
    
    local json = [[
    
        {
    
            "name": "御风小栈",
    
            "url": "http://www.x86.pub"
    
        }
    
    ]]
    
    local table = cjson.decode(json)
    
    print(table["name"], table["url"])  
  
---  
  
### table转为json字符串

    
    
    local cjson = require "cjson"
    
    local table = {}
    
    table["name"] = "御风小栈"
    
    table["url"] = "http://www.x86.pub"
    
    local json = cjson.encode(table)
    
    print(json)
